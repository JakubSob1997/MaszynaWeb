//Program NWD
//Zajduje nwd z liczb A i B
//Wynik zapisuje w AK

//A > B
aMinusb: POB a                
aMinusB+1: ODE b
SOZ stop  
SOM bMinusA   
LAD a 
SOB aMinusB+1

//B>A
bMinusA: POB b
bMinusA+1: ODE a
SOZ stop
SOM aMinusb
LAD b
SOB bMinusA+1

//A==B znaleziono NWD
stop: DOD a
STP

a: RST 120 //A=120
b: RST 70 //B=70
