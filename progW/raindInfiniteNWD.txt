//Program NWD
//Zajduje nwd z liczb A i B
//Wynik zapisuje w AK



getA: IN 6
SOM getA
SOZ getA
LAD a
getB: IN 6
SOM getB
SOZ getB
LAD b



OUT 5
POB a
OUT 4
POB b
OUT 4

//A > B
aMinusb: POB a                
aMinusB+1: ODE b
SOZ stop  
SOM bMinusA   
LAD a 
SOB aMinusB+1

//B>A
bMinusA: POB b
bMinusA+1: ODE a
SOZ stop
SOM aMinusb
LAD b
SOB bMinusA+1

//A==B znaleziono NWD
stop: DOD a
OUT 4
SOB 0

a: RST 120 //A=120
b: RST 70 //B=70
